Transforming ages to ranges

#Ages  category:value -> 1:0-12, 2:13-30, 3:31-60, 4:61-inf
data$Age[data$Age<=12] <- 1
data$Age[data$Age>12 & data$Age<=30] <- 2 
data$Age[data$Age>30 & data$Age<=60] <- 3 
data$Age[data$Age>60] <- 4
